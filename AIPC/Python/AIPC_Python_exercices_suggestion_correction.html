from scipy import optimize
from scipy.stats import norm
h=plt.hist(r2)
num=h[0]
centers=(h[1][1:]+h[1][0:-1])/2
def myg(x,k,x0,s0):
    return k*norm.pdf(x,x0,s0)

p1,p2=optimize.curve_fit(myg,centers,num,p0=[100,200,10])
